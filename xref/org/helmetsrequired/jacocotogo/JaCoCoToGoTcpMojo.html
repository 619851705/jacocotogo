28  /**
29   *  Retrieves jacoco data from remote servers where
30   *  the jacoco javaagent is running with output=tcpserver
31   *
32   * @author Matthew C. Jenkins
33   * @ since 1.0
34   */
35  @Mojo(name = "tcp")
36  public class JaCoCoToGoTcpMojo extends AbstractMojo {
37      /**
38       * Whether the build should be failed if JaCoCo execution data can not be fetched
39       */
40      @Parameter(required = true, property = "jacocotogo.failOnError", defaultValue="false")
41      private boolean failOnError;
42      /**
43       * The hostname where the jacoco javaagent is running
44       */
45      @Parameter(required = true, property = "jacocotogo.hostname")
46      private String hostname;
47      
48      /**
49       * The port where the tcpserver is listening
50       */
51      @Parameter(required = true, property = "jacocotogo.port")
52      int port;
53      
54      /**
55       * The file to write with the fetched jacoco data
56       */
57      @Parameter(required = true, property = "jacocotogo.outputFile", defaultValue = "${project.build.directory}/jacocotogo/jacoco.exec")
58      private String outputFile;
59      /**
60       * whether to reset the coverage statistics after fetching the jacoco data
61       */
62      @Parameter(property = "jacocotogo.resetAfterFetch", defaultValue = "true")
63      private boolean resetAfterFetch;
64  
65      /** {@inheritDoc} */
66      @Override
67      public void execute() throws MojoExecutionException, MojoFailureException {                                
68          File file = new File(outputFile);        
69          File directory = file.getAbsoluteFile().getParentFile();        
70          
71          if (!directory.exists()) {
72              getLog().debug("creating directory: " + directory.getAbsolutePath());
73              directory.mkdirs();
74          }
75          
76          try {            
77              JaCoCoToGo.fetchJaCoCoDataOverTcp(hostname, port, file, resetAfterFetch);
78          } catch (JaCoCoToGoException ex) {
79              getLog().warn("Exception while running plugin.  failOnError = " + failOnError + ". " + ex.getMessage());
80              if (failOnError) {
81                  throw new MojoExecutionException("Exception while running plugin", ex);
82              }
83          } catch (JaCoCoToGoValidationException ex) {
84              getLog().warn("Exception while running plugin.  failOnError = " + failOnError + ". " + ex.getMessage());
85              if (failOnError) {
86                  throw new MojoFailureException("Exception while running plugin", ex);
87              }
88          }       
89      }
90      
91      /**
92       * <p>Setter for the field <code>failOnError</code>.</p>
93       * 
94       * @param failOnError whether build should be failed if an error occurs
95       */
96      public void setFailOnError(boolean failOnError) {
97          this.failOnError = failOnError;
98      }

116     }

124         this.outputFile = outputFile;
125     }

133         this.resetAfterFetch = resetAfterFetch;
134     }
